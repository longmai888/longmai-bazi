## 一、数据库设计

### 1.1 用户表 (users)
```sql
CREATE TABLE users (
    id INT PRIMARY KEY AUTO_INCREMENT,
    
    -- 基本信息
    phone VARCHAR(20) UNIQUE COMMENT '手机号',
    email VARCHAR(100) UNIQUE COMMENT '邮箱',
    password_hash VARCHAR(255) COMMENT '密码',
    
    -- 微信登录
    wx_openid VARCHAR(100) UNIQUE COMMENT '微信openid',
    wx_unionid VARCHAR(100) COMMENT '微信unionid',
    wx_nickname VARCHAR(100) COMMENT '微信昵称',
    wx_avatar VARCHAR(255) COMMENT '微信头像',
    
    -- QQ登录
    qq_openid VARCHAR(100) UNIQUE COMMENT 'QQ openid',
    qq_nickname VARCHAR(100) COMMENT 'QQ昵称',
    qq_avatar VARCHAR(255) COMMENT 'QQ头像',
    
    -- 会员信息
    member_level TINYINT DEFAULT 0 COMMENT '0普通 1VIP 2SVIP',
    member_expire_at TIMESTAMP COMMENT '会员到期时间',
    
    -- 积分系统
    points INT DEFAULT 0 COMMENT '积分余额',
    total_points INT DEFAULT 0 COMMENT '累计积分',
    
    -- 账户状态
    status TINYINT DEFAULT 1 COMMENT '0禁用 1正常',
    last_login_at TIMESTAMP COMMENT '最后登录时间',
    last_login_ip VARCHAR(50) COMMENT '最后登录IP',
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    INDEX idx_phone (phone),
    INDEX idx_wx_openid (wx_openid),
    INDEX idx_member_level (member_level)
);
```

## 四、核心功能实现

### 4.1 会员权益检查
```javascript
// 检查用户是否有某项权益
async function checkUserBenefit(userId, benefitName) {
    const user = await db.queryOne(
        'SELECT member_level, member_expire_at FROM users WHERE id = ?',
        [userId]
    );
    
    // 检查会员是否过期
    if (user.member_expire_at && new Date() > new Date(user.member_expire_at)) {
        // 会员已过期，降级为普通会员
        await db.update('users', 
            { member_level: 0, member_expire_at: null },
            { id: userId }
        );
        user.member_level = 0;
    }
    
    // 获取等级权益
    const level = await db.queryOne(
        'SELECT benefits FROM member_levels WHERE level = ?',
        [user.member_level]
    );
    
    const benefits = JSON.parse(level.benefits);
    return benefits[benefitName] || false;
}

// 检查咨询次数
async function checkConsultCount(userId) {
    const user = await db.queryOne(
        'SELECT member_level FROM users WHERE id = ?',
        [userId]
    );
    
    const level = await db.queryOne(
        'SELECT benefits FROM member_levels WHERE level = ?',
        [user.member_level]
    );
    
    const benefits = JSON.parse(level.benefits);
    const totalCount = benefits.consult_count;
    
    if (totalCount === -1) {
        return { unlimited: true };  // 无限次
    }
    
    // 查询本月已使用次数
    const usedCount = await db.queryOne(`
        SELECT COUNT(*) as count FROM consult_orders 
        WHERE user_id = ? 
        AND status = 'completed'
        AND created_at >= DATE_FORMAT(NOW(), '%Y-%m-01')
    `, [userId]);
    
    return {
        unlimited: false,
        total: totalCount,
        used: usedCount.count,
        remaining: totalCount - usedCount.count
    };
}
```

### 4.2 积分系统
```javascript
// 增加积分
async function addPoints(userId, points, sourceType, sourceId, description) {
    const user = await db.queryOne(
        'SELECT points FROM users WHERE id = ?',
        [userId]
    );
    
    const newBalance = user.points + points;
    
    // 开启事务
    await db.transaction(async (trx) => {
        // 更新用户积分
        await trx.update('users', {
            points: newBalance,
            total_points: db.raw('total_points + ?', [points])
        }, { id: userId });
        
        // 记录积分日志
        await trx.insert('points_logs', {
            user_id: userId,
            type: 1,  // 获得
            points: points,
            balance: newBalance,
            source_type: sourceType,
            source_id: sourceId,
            description: description,
            created_at: new Date()
        });
    });
    
    return { success: true, new_balance: newBalance };
}

// 消耗积分
async function deductPoints(userId, points, description) {
    const user = await db.queryOne(
        'SELECT points FROM users WHERE id = ?',
        [userId]
    );
    
    if (user.points < points) {
        return { success: false, message: '积分不足' };
    }
    
    const newBalance = user.points - points;
    
    await db.transaction(async (trx) => {
        await trx.update('users', { points: newBalance }, { id: userId });
        
        await trx.insert('points_logs', {
            user_id: userId,
            type: 2,  // 消耗
            points: -points,
            balance: newBalance,
            description: description,
            created_at: new Date()
        });
    });
    
    return { success: true, new_balance: newBalance };
}
```

### 4.3 会员到期处理
```javascript
// 定时任务：每天检查过期会员
async function checkExpiredMembers() {
    const expiredUsers = await db.query(`
        SELECT id FROM users 
        WHERE member_level > 0 
        AND member_expire_at < CURDATE()
    `);
    
    for (const user of expiredUsers) {
        await db.update('users', {
            member_level: 0,
            member_expire_at: null
        }, { id: user.id });
        
        // 发送过期通知
        await sendNotification(user.id, {
            type: 'member_expired',
            title: '会员已过期',
            content: '您的VIP会员已过期，续费可继续享受权益'
        });
    }
    
    console.log(`处理了 ${expiredUsers.length} 个过期会员`);
}
```
