## 一、数据库设计

### 1.1 师傅表 (masters)
```sql
CREATE TABLE masters (
    id INT PRIMARY KEY AUTO_INCREMENT,
    user_id INT NOT NULL,
    real_name VARCHAR(50) NOT NULL COMMENT '真实姓名',
    phone VARCHAR(20) NOT NULL COMMENT '手机号',
    email VARCHAR(100) COMMENT '邮箱',
    avatar VARCHAR(255) COMMENT '头像URL',
    
    -- 资质信息
    id_card VARCHAR(18) COMMENT '身份证号',
    id_card_front VARCHAR(255) COMMENT '身份证正面',
    id_card_back VARCHAR(255) COMMENT '身份证反面',
    certificate VARCHAR(255) COMMENT '资质证书',
    certificate_no VARCHAR(50) COMMENT '证书编号',
    
    -- 专业信息
    specialties JSON COMMENT '擅长领域 ["八字", "紫微", "塔罗"]',
    experience_years INT COMMENT '从业年限',
    introduction TEXT COMMENT '个人简介',
    service_description TEXT COMMENT '服务介绍',
    
    -- 价格设置
    text_price DECIMAL(10,2) DEFAULT 99.00 COMMENT '文字咨询价格',
    voice_price DECIMAL(10,2) DEFAULT 199.00 COMMENT '语音咨询价格',
    video_price DECIMAL(10,2) DEFAULT 299.00 COMMENT '视频咨询价格',
    offline_price DECIMAL(10,2) DEFAULT 500.00 COMMENT '线下咨询价格',
    
    -- 状态
    status TINYINT DEFAULT 0 COMMENT '0待审核 1已通过 2已拒绝 3已禁用',
    reject_reason VARCHAR(255) COMMENT '拒绝原因',
    
    -- 评级
    star_rating DECIMAL(2,1) DEFAULT 5.0 COMMENT '星级 1-5',
    review_count INT DEFAULT 0 COMMENT '评价数量',
    service_count INT DEFAULT 0 COMMENT '服务次数',
    medal VARCHAR(50) COMMENT '勋章',
    
    -- 收入
    total_income DECIMAL(10,2) DEFAULT 0.00 COMMENT '总收入',
    available_balance DECIMAL(10,2) DEFAULT 0.00 COMMENT '可提现余额',
    
    -- 时间
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    INDEX idx_status (status),
    INDEX idx_specialties (specialties),
    INDEX idx_star_rating (star_rating)
);
```

## 四、核心功能实现

### 4.1 师傅注册流程
```javascript
// 师傅申请注册
async function applyMaster(formData) {
    // 1. 验证表单
    const validation = validateMasterForm(formData);
    if (!validation.valid) {
        return { code: 400, message: validation.error };
    }
    
    // 2. 检查是否已申请
    const existing = await db.query(
        'SELECT id FROM masters WHERE user_id = ?',
        [formData.user_id]
    );
    if (existing.length > 0) {
        return { code: 400, message: '您已提交申请，请勿重复提交' };
    }
    
    // 3. 保存数据
    const masterId = await db.insert('masters', {
        user_id: formData.user_id,
        real_name: formData.real_name,
        phone: formData.phone,
        // ... 其他字段
        status: 0,  // 待审核
        created_at: new Date()
    });
    
    // 4. 发送审核通知（给管理员）
    await sendAdminNotification({
        type: 'master_apply',
        master_id: masterId,
        message: `新师傅申请：${formData.real_name}`
    });
    
    return { code: 200, message: '申请已提交', data: { master_id: masterId } };
}
```

### 4.2 师傅评级计算
```javascript
// 计算师傅星级
function calculateMasterRating(masterId) {
    const reviews = db.query(
        'SELECT rating FROM reviews WHERE master_id = ?',
        [masterId]
    );
    
    if (reviews.length === 0) return 5.0;
    
    const avgRating = reviews.reduce((sum, r) => sum + r.rating, 0) / reviews.length;
    return Math.round(avgRating * 10) / 10;  // 保留1位小数
}

// 更新师傅勋章
function updateMasterMedal(masterId) {
    const master = db.queryOne('SELECT * FROM masters WHERE id = ?', [masterId]);
    
    let medal = null;
    if (master.service_count >= 100 && master.star_rating >= 4.8) {
        medal = '金牌师傅';
    } else if (master.service_count >= 50 && master.star_rating >= 4.5) {
        medal = '认证师傅';
    }
    
    db.update('masters', { medal }, { id: masterId });
}
```

### 4.3 收入统计
```javascript
// 获取师傅收入统计
async function getMasterIncome(masterId, startDate, endDate) {
    const stats = await db.query(`
        SELECT 
            DATE(created_at) as date,
            SUM(amount) as income,
            COUNT(*) as order_count
        FROM orders
        WHERE master_id = ? 
        AND status = 'completed'
        AND created_at BETWEEN ? AND ?
        GROUP BY DATE(created_at)
        ORDER BY date DESC
    `, [masterId, startDate, endDate]);
    
    const total = stats.reduce((sum, s) => sum + s.income, 0);
    
    return {
        total_income: total,
        daily_stats: stats
    };
}
```
