## 一、数据库设计

### 1.1 咨询订单表 (consult_orders)
```sql
CREATE TABLE consult_orders (
    id INT PRIMARY KEY AUTO_INCREMENT,
    order_no VARCHAR(50) UNIQUE NOT NULL COMMENT '订单编号 C202603210001',
    
    -- 关联信息
    user_id INT NOT NULL COMMENT '用户ID',
    master_id INT NOT NULL COMMENT '师傅ID',
    
    -- 咨询信息
    consult_type TINYINT NOT NULL COMMENT '1文字 2语音 3视频 4线下',
    price DECIMAL(10,2) NOT NULL COMMENT '咨询价格',
    
    -- 问题描述
    question TEXT COMMENT '咨询问题',
    attachments JSON COMMENT '附件 [图片URL]',
    
    -- 预约时间
    appoint_date DATE COMMENT '预约日期',
    appoint_time_start TIME COMMENT '开始时间',
    appoint_time_end TIME COMMENT '结束时间',
    
    -- 订单状态
    status TINYINT DEFAULT 0 COMMENT 
        '0待支付 1待确认 2待咨询 3咨询中 4待评价 5已完成 6已取消 7退款中 8已退款',
    
    -- 支付信息
    pay_amount DECIMAL(10,2) COMMENT '实付金额',
    pay_method VARCHAR(20) COMMENT '支付方式',
    pay_time TIMESTAMP COMMENT '支付时间',
    
    -- 平台抽成
    platform_fee DECIMAL(10,2) DEFAULT 0 COMMENT '平台抽成',
    master_income DECIMAL(10,2) DEFAULT 0 COMMENT '师傅收入',
    
    -- 时间记录
    confirm_time TIMESTAMP COMMENT '师傅确认时间',
    start_time TIMESTAMP COMMENT '咨询开始时间',
    end_time TIMESTAMP COMMENT '咨询结束时间',
    
    -- 取消/退款
    cancel_reason VARCHAR(255) COMMENT '取消原因',
    cancel_time TIMESTAMP COMMENT '取消时间',
    refund_amount DECIMAL(10,2) COMMENT '退款金额',
    refund_reason VARCHAR(255) COMMENT '退款原因',
    refund_time TIMESTAMP COMMENT '退款时间',
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    INDEX idx_user_id (user_id),
    INDEX idx_master_id (master_id),
    INDEX idx_status (status),
    INDEX idx_created_at (created_at)
);
```

## 四、核心功能实现

### 4.1 创建订单流程
```javascript
async function createConsultOrder(userId, params) {
    // 1. 检查用户会员权益
    const benefitCheck = await checkConsultCount(userId);
    if (!benefitCheck.unlimited && benefitCheck.remaining <= 0) {
        return { code: 400, message: '本月咨询次数已用完' };
    }
    
    // 2. 检查师傅是否可预约
    const isAvailable = await checkMasterAvailability(
        params.master_id, 
        params.appoint_date, 
        params.appoint_time_start
    );
    if (!isAvailable) {
        return { code: 400, message: '该时间段已被预约' };
    }
    
    // 3. 获取师傅价格
    const master = await db.queryOne(
        'SELECT text_price, voice_price, video_price, offline_price FROM masters WHERE id = ?',
        [params.master_id]
    );
    const priceField = ['text_price', 'voice_price', 'video_price', 'offline_price'][params.consult_type - 1];
    const price = master[priceField];
    
    // 4. 计算平台抽成（20%）
    const platformFee = price * 0.2;
    const masterIncome = price * 0.8;
    
    // 5. 生成订单号
    const orderNo = generateOrderNo();  // C202603210001
    
    // 6. 创建订单
    const orderId = await db.insert('consult_orders', {
        order_no: orderNo,
        user_id: userId,
        master_id: params.master_id,
        consult_type: params.consult_type,
        price: price,
        question: params.question,
        attachments: JSON.stringify(params.attachments || []),
        appoint_date: params.appoint_date,
        appoint_time_start: params.appoint_time_start,
        status: 0,  // 待支付
        platform_fee: platformFee,
        master_income: masterIncome,
        created_at: new Date()
    });
    
    // 7. 生成支付参数
    const payParams = await generatePayParams(orderNo, price);
    
    return {
        code: 200,
        data: {
            order_id: orderId,
            order_no: orderNo,
            price: price,
            pay_params: payParams
        }
    };
}
```

### 4.2 检查师傅可预约时间
```javascript
async function checkMasterAvailability(masterId, date, timeStart) {
    // 1. 获取师傅营业设置
    const weekDay = new Date(date).getDay();
    const schedule = await db.queryOne(
        'SELECT is_working, time_slots FROM master_schedules WHERE master_id = ? AND week_day = ?',
        [masterId, weekDay]
    );
    
    if (!schedule || !schedule.is_working) {
        return false;  // 当天不营业
    }
    
    // 2. 检查时间段是否在营业时间内
    const timeSlots = JSON.parse(schedule.time_slots);
    const isInWorkingHours = timeSlots.some(slot => 
        timeStart >= slot.start && timeStart < slot.end
    );
    if (!isInWorkingHours) {
        return false;
    }
    
    // 3. 检查是否已被预约
    const existingOrder = await db.queryOne(`
        SELECT id FROM consult_orders 
        WHERE master_id = ? 
        AND appoint_date = ? 
        AND appoint_time_start = ?
        AND status NOT IN (6, 8)  // 排除已取消和已退款
    `, [masterId, date, timeStart]);
    
    return !existingOrder;
}
```

### 4.3 订单状态流转
```javascript
// 订单状态机
const OrderStatus = {
    PENDING_PAY: 0,      // 待支付
    PENDING_CONFIRM: 1,  // 待确认（师傅）
    PENDING_CONSULT: 2,  // 待咨询
    CONSULTING: 3,       // 咨询中
    PENDING_REVIEW: 4,   // 待评价
    COMPLETED: 5,        // 已完成
    CANCELLED: 6,        // 已取消
    REFUNDING: 7,        // 退款中
    REFUNDED: 8          // 已退款
};

// 状态流转
const statusFlow = {
    [OrderStatus.PENDING_PAY]: ['pay', 'cancel'],
    [OrderStatus.PENDING_CONFIRM]: ['confirm', 'reject', 'cancel'],
    [OrderStatus.PENDING_CONSULT]: ['start', 'cancel'],
    [OrderStatus.CONSULTING]: ['end'],
    [OrderStatus.PENDING_REVIEW]: ['review'],
    [OrderStatus.COMPLETED]: [],
    [OrderStatus.CANCELLED]: [],
    [OrderStatus.REFUNDING]: ['approve_refund', 'reject_refund'],
    [OrderStatus.REFUNDED]: []
};

async function updateOrderStatus(orderId, action, operator) {
    const order = await db.queryOne('SELECT * FROM consult_orders WHERE id = ?', [orderId]);
    
    // 检查操作是否允许
    if (!statusFlow[order.status].includes(action)) {
        return { code: 400, message: '当前状态不允许此操作' };
    }
    
    let newStatus = order.status;
    let updateData = {};
    
    switch (action) {
        case 'pay':
            newStatus = OrderStatus.PENDING_CONFIRM;
            updateData.pay_time = new Date();
            // 通知师傅有新订单
            await notifyMaster(order.master_id, '新预约订单');
            break;
            
        case 'confirm':
            newStatus = OrderStatus.PENDING_CONSULT;
            updateData.confirm_time = new Date();
            // 通知用户订单已确认
            await notifyUser(order.user_id, '师傅已确认预约');
            break;
            
        case 'start':
            newStatus = OrderStatus.CONSULTING;
            updateData.start_time = new Date();
            break;
            
        case 'end':
            newStatus = OrderStatus.PENDING_REVIEW;
            updateData.end_time = new Date();
            // 结算师傅收入
            await settleMasterIncome(orderId);
            break;
            
        case 'review':
            newStatus = OrderStatus.COMPLETED;
            // 更新师傅评分
            await updateMasterRating(order.master_id);
            break;
            
        case 'cancel':
            newStatus = OrderStatus.CANCELLED;
            updateData.cancel_time = new Date();
            break;
    }
    
    await db.update('consult_orders', {
        status: newStatus,
        ...updateData
    }, { id: orderId });
    
    return { code: 200, message: '操作成功' };
}
```

### 4.4 即时通讯集成
```javascript
// 集成腾讯云IM或环信IM
const TIM = require('tim-js-sdk');

// 初始化IM
const tim = TIM.create({
    SDKAppID: 1400000000  // 腾讯云IM的SDKAppID
});

// 用户登录IM
async function loginIM(userId, userSig) {
    await tim.login({
        userID: String(userId),
        userSig: userSig
    });
}

// 创建咨询房间
async function createConsultRoom(orderId, userId, masterId) {
    const roomId = `consult_${orderId}`;
    
    // 创建群组
    await tim.createGroup({
        type: TIM.TYPES.GRP_CHATROOM,  // 聊天室
        name: `咨询-${orderId}`,
        groupID: roomId,
        memberList: [
            { userID: String(userId) },
            { userID: String(masterId) }
        ]
    });
    
    return roomId;
}

// 发送消息
async function sendMessage(roomId, message) {
    const messageObj = tim.createTextMessage({
        to: roomId,
        conversationType: TIM.TYPES.CONV_GROUP,
        payload: { text: message }
    });
    
    await tim.sendMessage(messageObj);
}
```
