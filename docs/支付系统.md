## 一、数据库设计

### 1.1 支付订单表 (payment_orders)
```sql
CREATE TABLE payment_orders (
    id INT PRIMARY KEY AUTO_INCREMENT,
    order_no VARCHAR(50) UNIQUE NOT NULL COMMENT '支付订单号 P202603210001',
    
    -- 关联业务订单
    biz_type VARCHAR(20) NOT NULL COMMENT '业务类型：consult/member/goods',
    biz_id INT NOT NULL COMMENT '业务订单ID',
    biz_order_no VARCHAR(50) COMMENT '业务订单编号',
    
    -- 支付信息
    user_id INT NOT NULL,
    amount DECIMAL(10,2) NOT NULL COMMENT '支付金额',
    pay_method VARCHAR(20) NOT NULL COMMENT 'wechat/alipay',
    
    -- 支付状态
    status TINYINT DEFAULT 0 COMMENT '0待支付 1支付中 2已支付 3已关闭 4已退款',
    
    -- 第三方支付信息
    third_order_no VARCHAR(100) COMMENT '第三方订单号',
    third_prepay_id VARCHAR(100) COMMENT '微信prepay_id',
    pay_time TIMESTAMP COMMENT '支付完成时间',
    
    -- 退款信息
    refund_amount DECIMAL(10,2) DEFAULT 0 COMMENT '已退款金额',
    refund_time TIMESTAMP COMMENT '退款时间',
    
    -- 过期时间
    expire_at TIMESTAMP COMMENT '订单过期时间',
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    INDEX idx_biz (biz_type, biz_id),
    INDEX idx_user_id (user_id),
    INDEX idx_status (status),
    INDEX idx_order_no (order_no)
);
```

### 1.2 退款记录表 (refund_records)
```sql
CREATE TABLE refund_records (
    id INT PRIMARY KEY AUTO_INCREMENT,
    refund_no VARCHAR(50) UNIQUE NOT NULL COMMENT '退款单号 R202603210001',
    payment_order_id INT NOT NULL,
    
    -- 退款信息
    amount DECIMAL(10,2) NOT NULL COMMENT '退款金额',
    reason VARCHAR(255) COMMENT '退款原因',
    
    -- 退款状态
    status TINYINT DEFAULT 0 COMMENT '0申请中 1处理中 2已退款 3已拒绝',
    
    -- 第三方退款信息
    third_refund_no VARCHAR(100) COMMENT '第三方退款单号',
    refund_time TIMESTAMP COMMENT '退款完成时间',
    
    -- 审核信息
    operator_id INT COMMENT '操作员ID',
    remark VARCHAR(255) COMMENT '审核备注',
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    INDEX idx_payment_order (payment_order_id),
    INDEX idx_status (status)
);
```

### 1.3 财务流水表 (financial_records)
```sql
CREATE TABLE financial_records (
    id INT PRIMARY KEY AUTO_INCREMENT,
    
    -- 流水类型
    type TINYINT NOT NULL COMMENT '1收入 2支出 3退款',
    biz_type VARCHAR(20) COMMENT '业务类型',
    biz_id INT COMMENT '业务ID',
    
    -- 金额
    amount DECIMAL(10,2) NOT NULL,
    balance DECIMAL(10,2) NOT NULL COMMENT '变动后余额',
    
    -- 关联信息
    user_id INT COMMENT '用户ID（收入时）',
    master_id INT COMMENT '师傅ID（支出时）',
    
    -- 描述
    description VARCHAR(255) COMMENT '描述',
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    INDEX idx_type (type),
    INDEX idx_biz (biz_type, biz_id),
    INDEX idx_created_at (created_at)
);
```

## 三、核心功能实现

### 3.1 微信支付集成
```javascript
const wxpay = require('wechatpay-node-v3');

// 初始化微信支付
const wxPayClient = new wxpay({
    appid: 'wx1234567890',
    mchid: '1234567890',
    private_key: fs.readFileSync('./apiclient_key.pem'),
    cert_sn: '...',
    api_v3_key: 'your_api_v3_key'
});

// 创建支付订单
async function createWechatPayOrder(params) {
    const { paymentOrderNo, amount, description, openid } = params;
    
    // 调用微信支付统一下单接口
    const result = await wxPayClient.transactions_jsapi({
        description: description,
        out_trade_no: paymentOrderNo,
        notify_url: 'https://your-domain.com/api/payment/wechat-callback',
        amount: {
            total: Math.round(amount * 100)  // 转为分
        },
        payer: {
            openid: openid
        }
    });
    
    // 生成前端支付参数
    const payParams = {
        appId: result.appid,
        timeStamp: String(Math.floor(Date.now() / 1000)),
        nonceStr: generateNonceStr(),
        package: `prepay_id=${result.prepay_id}`,
        signType: 'RSA'
    };
    
    // 计算签名
    payParams.paySign = generatePaySign(payParams);
    
    return {
        prepay_id: result.prepay_id,
        pay_params: payParams
    };
}

// 处理支付回调
async function handleWechatCallback(xmlData) {
    // 解密回调数据
    const decryptedData = decryptCallbackData(xmlData);
    
    const { out_trade_no, transaction_id, result_code } = decryptedData;
    
    if (result_code === 'SUCCESS') {
        // 更新支付订单状态
        await db.update('payment_orders', {
            status: 2,  // 已支付
            third_order_no: transaction_id,
            pay_time: new Date()
        }, { order_no: out_trade_no });
        
        // 更新业务订单状态
        const paymentOrder = await db.queryOne(
            'SELECT biz_type, biz_id FROM payment_orders WHERE order_no = ?',
            [out_trade_no]
        );
        
        await updateBizOrderStatus(paymentOrder.biz_type, paymentOrder.biz_id, 'paid');
        
        // 记录财务流水
        await recordFinancialTransaction({
            type: 1,  // 收入
            biz_type: paymentOrder.biz_type,
            biz_id: paymentOrder.biz_id,
            amount: decryptedData.total_fee / 100,
            description: `用户支付-${paymentOrder.biz_type}`
        });
    }
    
    return '<xml><return_code><![CDATA[SUCCESS]]></return_code></xml>';
}
```

### 3.3 退款处理
```javascript
// 申请退款
async function applyRefund(paymentOrderNo, refundAmount, reason) {
    const paymentOrder = await db.queryOne(
        'SELECT * FROM payment_orders WHERE order_no = ?',
        [paymentOrderNo]
    );
    
    // 检查是否可退款
    if (paymentOrder.status !== 2) {
        return { code: 400, message: '订单未支付，无法退款' };
    }
    
    if (paymentOrder.refund_amount + refundAmount > paymentOrder.amount) {
        return { code: 400, message: '退款金额超过支付金额' };
    }
    
    // 生成退款单号
    const refundNo = generateRefundNo();
    
    // 创建退款记录
    await db.insert('refund_records', {
        refund_no: refundNo,
        payment_order_id: paymentOrder.id,
        amount: refundAmount,
        reason: reason,
        status: 0,  // 申请中
        created_at: new Date()
    });
    
    // 如果是自动退款（小额），直接处理
    if (refundAmount <= 100) {
        await processRefund(refundNo);
    }
    
    return {
        code: 200,
        data: {
            refund_no: refundNo,
            status: 0
        }
    };
}

// 处理退款（调用微信/支付宝接口）
async function processRefund(refundNo) {
    const refundRecord = await db.queryOne(
        'SELECT * FROM refund_records WHERE refund_no = ?',
        [refundNo]
    );
    
    const paymentOrder = await db.queryOne(
        'SELECT * FROM payment_orders WHERE id = ?',
        [refundRecord.payment_order_id]
    );
    
    let result;
    
    if (paymentOrder.pay_method === 'wechat') {
        // 微信退款
        result = await wxPayClient.refund({
            out_trade_no: paymentOrder.order_no,
            out_refund_no: refundNo,
            amount: {
                refund: Math.round(refundRecord.amount * 100),
                total: Math.round(paymentOrder.amount * 100)
            }
        });
    } else {
        // 支付宝退款
        result = await alipaySdk.exec('alipay.trade.refund', {
            bizContent: {
                out_trade_no: paymentOrder.order_no,
                refund_amount: refundRecord.amount,
                out_request_no: refundNo
            }
        });
    }
    
    // 更新退款记录
    await db.update('refund_records', {
        status: 2,  // 已退款
        third_refund_no: result.refund_id || result.trade_no,
        refund_time: new Date()
    }, { refund_no: refundNo });
    
    // 更新支付订单
    await db.update('payment_orders', {
        refund_amount: db.raw('refund_amount + ?', [refundRecord.amount]),
        refund_time: new Date()
    }, { id: paymentOrder.id });
    
    // 记录财务流水
    await recordFinancialTransaction({
        type: 3,  // 退款
        biz_type: paymentOrder.biz_type,
        biz_id: paymentOrder.biz_id,
        amount: -refundRecord.amount,
        description: `退款-${refundNo}`
    });
    
    return { success: true };
}
```

### 3.4 订单超时处理
```javascript
// 定时任务：关闭超时未支付订单
async function closeExpiredOrders() {
    const expiredOrders = await db.query(`
        SELECT id, order_no FROM payment_orders 
        WHERE status = 0 
        AND expire_at < NOW()
    `);
    
    for (const order of expiredOrders) {
        // 更新支付订单状态
        await db.update('payment_orders', {
            status: 3  // 已关闭
        }, { id: order.id });
        
        // 更新业务订单状态
        const paymentOrder = await db.queryOne(
            'SELECT biz_type, biz_id FROM payment_orders WHERE id = ?',
            [order.id]
        );
        
        await updateBizOrderStatus(paymentOrder.biz_type, paymentOrder.biz_id, 'cancelled');
        
        console.log(`关闭超时订单: ${order.order_no}`);
    }
}

// 每5分钟执行一次
setInterval(closeExpiredOrders, 5 * 60 * 1000);
```

### 4.1 方案说明
- **方式**：个人微信/支付宝收款码

- **流程**：用户扫码支付 → 截图上传 → 人工确认 → 开通服务

```javascript
// 私域支付流程
async function privatePaymentFlow(orderId) {
    const order = await getOrderInfo(orderId);
    
    return {
        code: 200,
        data: {
            payment_method: 'private',
            amount: order.amount,
            qr_code_url: 'https://your-domain.com/qrcode/personal-wechat.jpg',
            instructions: [
                '1. 保存上方收款码',
                '2. 打开微信扫一扫',
                '3. 支付金额：¥' + order.amount,
                '4. 支付完成后截图上传'

// 上传支付凭证
async function uploadPaymentProof(orderId, imageUrl) {
    await db.update('orders', {
        payment_proof: imageUrl,
        status: 'pending_confirm',  // 待人工确认
        payment_proof_uploaded_at: new Date()
    }, { id: orderId });
    
    // 通知管理员审核
    await notifyAdmin({
        type: 'payment_proof_uploaded',
        order_id: orderId,
        message: `订单 ${orderId} 已上传支付凭证，请审核`
    });
    
    return { code: 200, message: '凭证已上传，等待审核' };
}

// 管理员确认收款
async function confirmPrivatePayment(orderId, adminId) {
    await db.update('orders', {
        status: 'paid',
        confirmed_by: adminId,
        confirmed_at: new Date()
    }, { id: orderId });
    
    // 开通服务
    await activateService(orderId);
    
    return { code: 200, message: '收款已确认，服务已开通' };
}
```
